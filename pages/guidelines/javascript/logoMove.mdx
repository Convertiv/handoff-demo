## Overview


```html title="slider.js"
<script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
<script type="text/javascript">
<script>
/**
 * This script is used to initialize the slider on the homepage, community, and
 * logo pages. It depends on jQuery.
 */
function logoMove(items, scrollSpace) {
  $(window).on("load scroll resize", function () {
    const $parallaxElement = items;
    $parallaxElement.each(function (i) {
      const $currentElement = $(this),
        windowTop = $(window).scrollTop(),
        elementTop = $currentElement.offset().top,
        elementHeight = $currentElement.height(),
        viewPortHeight = window.innerHeight - elementHeight,
        scrolled = windowTop - elementTop + viewPortHeight;

      if (i === 1) {
        $currentElement.css({
          transform: "translate3d(" + scrolled * scrollSpace + "px, 0, 0)",
        });
      }
      if (i === 0) {
        $currentElement.css({
          transform: "translate3d(" + scrolled * -scrollSpace + "px, 0, 0)",
        });
      }
    });
  });
}
</script>
```

## Example Usage

```javascript title="Use the logoMove function"
import $ from "jquery";
import logoMove from "./shared/logoMove";
// Bind the slider when the document is ready

```
